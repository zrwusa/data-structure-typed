import type { ElementCallback, TrieOptions } from '../../types';

/**
 * Single-character trie node used by {@link Trie}.
 * Each node stores a character key, a map of child nodes, and an end-of-word flag.
 */
export class TrieNode {
  constructor(key: string) {
    this._key = key;
    this._isEnd = false;
    this._children = new Map<string, TrieNode>();
  }
 
  protected _key: string;
 
  /** Returns the character stored at this node (root is an empty string). */
  get key(): string { return this._key; }
 
  /** Sets the character stored at this node. */
  set key(value: string) { this._key = value; }
 
  protected _children: Map<string, TrieNode>;
 
  /** Child map keyed by next character. */
  get children(): Map<string, TrieNode> { return this._children; }
 
  /** Replaces the child map for this node. */
  set children(value: Map<string, TrieNode>) { this._children = value; }
 
  protected _isEnd: boolean;
 
  /** Whether this node terminates a stored word. */
  get isEnd(): boolean { return this._isEnd; }
 
  /** Marks/unmarks this node as a word terminator. */
  set isEnd(value: boolean) { this._isEnd = value; }
}
 
/**
 * Trie (prefix tree) for strings with optional case sensitivity and raw-element conversion.
 *
 * Key features:
 * - Add / delete / membership tests in O(l) where l is input length.
 * - Prefix checks (pure prefix and general prefix).
 * - Enumerate words by prefix via {@link getWords}.
 * - Iterable over all stored words (DFS order).
 *
 * @template R Raw element type accepted by {@link TrieOptions.toElementFn | toElementFn}.
 * @example
 * // Autocomplete: Prefix validation and checking
 *     const autocomplete = new Trie<string>(['gmail.com', 'gmail.co.nz', 'gmail.co.jp', 'yahoo.com', 'outlook.com']);
 *
 *     // Get all completions for a prefix
 *     const gmailCompletions = autocomplete.getWords('gmail');
 *     console.log(gmailCompletions); // ['gmail.com', 'gmail.co.nz', 'gmail.co.jp']
 * @example
 * // File System Path Operations
 *     const fileSystem = new Trie<string>([
 *       '/home/user/documents/file1.txt',
 *       '/home/user/documents/file2.txt',
 *       '/home/user/pictures/photo.jpg',
 *       '/home/user/pictures/vacation/',
 *       '/home/user/downloads'
 *     ]);
 *
 *     // Find common directory prefix
 *     console.log(fileSystem.getLongestCommonPrefix()); // '/home/user/'
 *
 *     // List all files in a directory
 *     const documentsFiles = fileSystem.getWords('/home/user/documents/');
 *     console.log(documentsFiles); // ['/home/user/documents/file1.txt', '/home/user/documents/file2.txt']
 * @example
 * // Autocomplete: Basic word suggestions
 *     // Create a trie for autocomplete
 *     const autocomplete = new Trie<string>([
 *       'function',
 *       'functional',
 *       'functions',
 *       'class',
 *       'classes',
 *       'classical',
 *       'closure',
 *       'const',
 *       'constructor'
 *     ]);
 *
 *     // Test autocomplete with different prefixes
 *     console.log(autocomplete.getWords('fun')); // ['functional', 'functions', 'function']
 *     console.log(autocomplete.getWords('cla')); // ['classes', 'classical', 'class']
 *     console.log(autocomplete.getWords('con')); // ['constructor', 'const']
 *
 *     // Test with non-matching prefix
 *     console.log(autocomplete.getWords('xyz')); // []
 * @example
 * // Dictionary: Case-insensitive word lookup
 *     // Create a case-insensitive dictionary
 *     const dictionary = new Trie<string>([], { caseSensitive: false });
 *
 *     // Add words with mixed casing
 *     dictionary.add('Hello');
 *     dictionary.add('WORLD');
 *     dictionary.add('JavaScript');
 *
 *     // Test lookups with different casings
 *     console.log(dictionary.has('hello')); // true
 *     console.log(dictionary.has('HELLO')); // true
 *     console.log(dictionary.has('Hello')); // true
 *     console.log(dictionary.has('javascript')); // true
 *     console.log(dictionary.has('JAVASCRIPT')); // true
 * @example
 * // IP Address Routing Table
 *     // Add IP address prefixes and their corresponding routes
 *     const routes = {
 *       '192.168.1': 'LAN_SUBNET_1',
 *       '192.168.2': 'LAN_SUBNET_2',
 *       '10.0.0': 'PRIVATE_NETWORK_1',
 *       '10.0.1': 'PRIVATE_NETWORK_2'
 *     };
 *
 *     const ipRoutingTable = new Trie<string>(Object.keys(routes));
 *
 *     // Check IP address prefix matching
 *     console.log(ipRoutingTable.hasPrefix('192.168.1')); // true
 *     console.log(ipRoutingTable.hasPrefix('192.168.2')); // true
 *
 *     // Validate IP address belongs to subnet
 *     const ip = '192.168.1.100';
 *     const subnet = ip.split('.').slice(0, 3).join('.');
 *     console.log(ipRoutingTable.hasPrefix(subnet)); // true
 */
export class Trie<R = any> extends IterableElementBase<string, R> {
  /**
   * Creates a trie and optionally seeds it with `words`.
   * @param {Iterable<string> | Iterable<R>} [words=[]] Words or raw elements to insert.
   * @param {TrieOptions<R>} [options] Optional configuration: `{ caseSensitive, toElementFn }`.
   * @remarks Complexity  Time: O(k * l) to add k words of average length l; Space: O(k * l).
   */
  constructor(words: Iterable<string> | Iterable<R> = [], options?: TrieOptions<R>) {
    super(options);
    if (options) {
      const { caseSensitive } = options;
      if (caseSensitive !== undefined) this._caseSensitive = caseSensitive;
    }
    if (words) this.addMany(words);
  }
 
  protected _size = 0;
 
  /** Number of words stored in the trie. @remarks Complexity  Time: O(1); Space: O(1). */
  get size(): number { return this._size; }
 
  protected _caseSensitive = true;
 
  /** Whether lookups are case-sensitive. @remarks Complexity  Time: O(1); Space: O(1). */
  get caseSensitive(): boolean { return this._caseSensitive; }
 
  protected _root: TrieNode = new TrieNode('');
 
  /** Root node (represents empty prefix). @remarks Complexity  Time: O(1); Space: O(1). */
  get root(): TrieNode { return this._root; }
 
  /**
   * Adds a word to the trie.
   * @param {string} word Word to add.
   * @returns {boolean} `true` if inserted (word was new); otherwise `false`.
   * @remarks Complexity  Time: O(l); Space: O(l).
   */
  add(word: string): boolean {
    word = this._caseProcess(word);
    let cur = this.root;
    let isNewWord = false;
    for (const c of word) {
      let nodeC = cur.children.get(c);
      if (!nodeC) {
        nodeC = new TrieNode(c);
        cur.children.set(c, nodeC);
      }
      cur = nodeC;
    }
    if (!cur.isEnd) {
      isNewWord = true;
      cur.isEnd = true;
      this._size += 1;
    }
    return isNewWord;
  }
 
  /**
   * Adds many words/raw elements.
   * @param {Iterable<string> | Iterable<R>} words Words or raw elements.
   * @returns {boolean[]} Per-word insertion results from {@link add}.
   * @remarks Complexity  Time: O(k * l); Space: O(k * l).
   */
  addMany(words: Iterable<string> | Iterable<R>): boolean[] {
    const ans: boolean[] = [];
    for (const word of words) {
      if (this.toElementFn) ans.push(this.add(this.toElementFn(word as R)));
      else ans.push(this.add(word as string));
    }
    return ans;
  }
 
  /**
   * Checks whether a word exists in the trie.
   * @param {string} word Word to check.
   * @returns {boolean}
   * @remarks Complexity  Time: O(l); Space: O(1).
   */
  override has(word: string): boolean {
    word = this._caseProcess(word);
    let cur = this.root;
    for (const c of word) {
      const nodeC = cur.children.get(c);
      if (!nodeC) return false;
      cur = nodeC;
    }
    return cur.isEnd;
  }
 
  /** Returns whether the trie is empty. @remarks Complexity  Time: O(1); Space: O(1). */
  isEmpty(): boolean { return this._size === 0; }
 
  /** Clears all words and resets the root. @remarks Complexity  Time: O(1); Space: O(1). */
  clear(): void { this._size = 0; this._root = new TrieNode(''); }
 
  /**
   * Deletes a word from the trie.
   * @param {string} word Word to remove.
   * @returns {boolean} `true` if removed; otherwise `false`.
   * @remarks Complexity  Time: O(l); Space: O(l) due to recursion depth.
   */
  delete(word: string): boolean {
    word = this._caseProcess(word);
    let isDeleted = false;
    const dfs = (cur: TrieNode, i: number): boolean => {
      const char = word[i];
      const child = cur.children.get(char);
      if (child) {
        if (i === word.length - 1) {
          if (child.isEnd) {
            if (child.children.size > 0) child.isEnd = false;
            else cur.children.delete(char);
            isDeleted = true;
            return true;
          }
          return false;
        }
        const res = dfs(child, i + 1);
        if (res && !cur.isEnd && child.children.size === 0) {
          cur.children.delete(char);
          return true;
        }
        return false;
      }
      return false;
    };
 
    dfs(this.root, 0);
    if (isDeleted) this._size -= 1;
    return isDeleted;
  }
 
  /**
   * Computes the height (maximum depth) of the trie.
   * @returns {number}
   * @remarks Complexity  Time: O(n); Space: O(n) in worst case due to recursion.
   */
  getHeight(): number {
    const startNode = this.root;
    let maxDepth = 0;
    if (startNode) {
      const bfs = (node: TrieNode, level: number) => {
        if (level > maxDepth) maxDepth = level;
        const { children } = node;
        if (children) for (const child of children.entries()) bfs(child[1], level + 1);
      };
      bfs(startNode, 0);
    }
    return maxDepth;
  }
 
  /**
   * Returns `true` if `input` is a stored prefix but not a complete word.
   * @param {string} input Input string to test.
   * @returns {boolean}
   * @remarks Complexity  Time: O(l); Space: O(1).
   */
  hasPurePrefix(input: string): boolean {
    input = this._caseProcess(input);
    let cur = this.root;
    for (const c of input) {
      const nodeC = cur.children.get(c);
      if (!nodeC) return false;
      cur = nodeC;
    }
    return !cur.isEnd;
  }
 
  /**
   * Returns `true` if `input` is a prefix of at least one stored word.
   * @param {string} input Input prefix.
   * @returns {boolean}
   * @remarks Complexity  Time: O(l); Space: O(1).
   */
  hasPrefix(input: string): boolean {
    input = this._caseProcess(input);
    let cur = this.root;
    for (const c of input) {
      const nodeC = cur.children.get(c);
      if (!nodeC) return false;
      cur = nodeC;
    }
    return true;
  }
 
  /**
   * Checks whether `input` is a common prefix (shared by all words).
   * @param {string} input Input prefix to test.
   * @returns {boolean}
   * @remarks Complexity  Time: O(n); Space: O(l).
   */
  hasCommonPrefix(input: string): boolean {
    input = this._caseProcess(input);
    let commonPre = '';
    const dfs = (cur: TrieNode) => {
      commonPre += cur.key;
      if (commonPre === input) return;
      if (cur.isEnd) return;
      if (cur && cur.children && cur.children.size === 1) dfs(Array.from(cur.children.values())[0]);
      else return;
    };
    dfs(this.root);
    return commonPre === input;
  }
 
  /**
   * Returns the longest common prefix among all stored words.
   * @returns {string}
   * @remarks Complexity  Time: O(n); Space: O(l).
   */
  getLongestCommonPrefix(): string {
    let commonPre = '';
    const dfs = (cur: TrieNode) => {
      commonPre += cur.key;
      if (cur.isEnd) return;
      if (cur && cur.children && cur.children.size === 1) dfs(Array.from(cur.children.values())[0]);
      else return;
    };
    dfs(this.root);
    return commonPre;
  }
 
  /**
   * Returns all words with a given prefix.
   * @param {string} [prefix=''] Prefix to match.
   * @param {number} [max=Number.MAX_SAFE_INTEGER] Maximum number of words to return.
   * @param {boolean} [isAllWhenEmptyPrefix=false] If `true`, return all words when `prefix` is empty.
   * @returns {string[]} Matching words (order depends on insertion/traversal).
   * @remarks Complexity  Time: O(w * l); Space: O(w * l).
   */
  getWords(prefix = '', max = Number.MAX_SAFE_INTEGER, isAllWhenEmptyPrefix = false): string[] {
    prefix = this._caseProcess(prefix);
    const words: string[] = [];
    let found = 0;
 
    function dfs(node: TrieNode, word: string) {
      for (const char of node.children.keys()) {
        const charNode = node.children.get(char);
        if (charNode !== undefined) dfs(charNode, word.concat(char));
      }
      if (node.isEnd) {
        if (found > max - 1) return;
        words.push(word);
        found += 1;
      }
    }
 
    let startNode = this.root;
    if (prefix) {
      for (const c of prefix) {
        const nodeC = startNode.children.get(c);
        if (nodeC) startNode = nodeC; else return [];
      }
    }
 
    if (isAllWhenEmptyPrefix || startNode !== this.root) dfs(startNode, prefix);
    return words;
  }
 
  /** Creates a shallow copy. @returns {Trie<R>} @remarks Complexity  Time: O(n); Space: O(n). */
  clone(): Trie<R> { return new Trie<R>(this, { caseSensitive: this.caseSensitive, toElementFn: this.toElementFn }); }
 
  /**
   * Filters words by a predicate.
   * @param {ElementCallback<string, R, boolean>} predicate Invoked as `(word, index, trie)`.
   * @param {any} [thisArg] Value of `this` in `predicate`.
   * @returns {Trie<R>} A new trie containing matching words.
   * @remarks Complexity  Time: O(n); Space: O(n).
   */
  filter(predicate: ElementCallback<string, R, boolean>, thisArg?: any): Trie<R> {
    const results = new Trie<R>([], { toElementFn: this.toElementFn, caseSensitive: this.caseSensitive });
    let index = 0;
    for (const word of this) {
      if (predicate.call(thisArg, word, index, this)) results.add(word);
      index += 1;
    }
    return results;
  }
 
  /**
   * Maps words using `callback` and returns a new trie of mapped values.
   * @template RM Raw input type for the mapped trie.
   * @param {ElementCallback<string, R, string>} callback Invoked as `(word, index, trie)`.
   * @param {(rawElement: RM) => string} [toElementFn] Optional converter for the mapped trie.
   * @param {any} [thisArg] Value of `this` for `callback`.
   * @returns {Trie<RM>} A new trie with mapped words.
   * @remarks Complexity  Time: O(n); Space: O(n).
   */
  map<RM>(
    callback: ElementCallback<string, R, string>,
    toElementFn?: (rawElement: RM) => string,
    thisArg?: any
  ): Trie<RM> {
    const newTrie = new Trie<RM>([], { toElementFn, caseSensitive: this.caseSensitive });
    let index = 0;
    for (const word of this) {
      newTrie.add(callback.call(thisArg, word, index, this));
      index += 1;
    }
    return newTrie;
  }
 
  /** Depth-first forward iterator over stored words. @remarks Complexity  Time: O(n); Space: O(h) recursion depth. */
  protected *_getIterator(): IterableIterator<string> {
    function* _dfs(node: TrieNode, path: string): IterableIterator<string> {
      if (node.isEnd) yield path;
      for (const [char, childNode] of node.children) {
        yield* _dfs(childNode, path + char);
      }
    }
    yield* _dfs(this.root, '');
  }
 
  /** Total elements (for {@link IterableElementBase}). */
  protected get _total() { return this._size; }
 
  /**
   * Normalizes casing according to {@link caseSensitive}.
   * @param {string} str Input string.
   * @returns {string} Normalized string.
   * @remarks Complexity  Time: O(l); Space: O(1).
   * @protected
   */
  protected _caseProcess(str: string): string {
    if (!this._caseSensitive) str = str.toLowerCase();
    return str;
  }
}
